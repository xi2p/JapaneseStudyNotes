\chapter{標準日本語　第18課}

摘要：\textbf{变得…}

\separatorline

\section{文法}

\subsection{「なる」表（自发）变得}
\begin{codeblock}
一类形く + なります \\
二类形に + なります \\
名词に 　+ なります
\end{codeblock}
表示自然变化或状态转变。

\begin{example}
    \begin{itemize}
        \item 北海道は　10月ごろから　寒\textbf{く　なります}。（北海道从十月左右开始变冷。）
        \item 操作は　簡単\textbf{に　なりました}。
        \item 王さんは　4月から　先生\textbf{に　なります}。（王先生四月开始就是老师了。）
    \end{itemize}
\end{example}

\separatorline

\subsection{「する」表使变得}
\begin{codeblock}
一类形く + します \\
二类形に + します \\
名词に 　+ します
\end{codeblock}
表示人为地\textbf{使}某物发生变化。强调人的主观意志和动作。

\begin{example}
    \begin{itemize}
        \item テレビの音を　大き\textbf{く　します}。（把电视音量调大一些）
        \item 部屋を　きれい\textbf{に　してください}。
        \item 田中さんの　子供を　看護婦\textbf{に　したいです}。（我想让田中先生的孩子当护士。）
    \end{itemize}
\end{example}
